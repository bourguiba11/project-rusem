Éléments HTML5

Éléments de 1er niveau

<html>

Représente la racine d'un document HTML.

Exemple
<html lang="fr">
  <head>...</head>
  <body>...</body>
</html>
<head>

Conteneur de métadonnées placé entre <html> et <body>.

Exemple
<head>
  <title>Mon site</title>
  <meta charset="UTF-8">
</head>
<body>

Définit le corps du document — tout le contenu visible.

Exemple
<body>
  <h1>Bonjour !</h1>
  <p>Bienvenue.</p>
</body>

Transitions et Animations

transition

Super-propriété: combine transition-property, transition-duration, transition-delay.

button {
  transition: background 0.3s ease;
}
button:hover {
  background: blue;
}
/* Plusieurs transitions */
div {
  transition: color 0.2s, width 0.5s 1s;
}
animation

Super-propriété d'animation. Nécessite un bloc @keyframes.

@keyframes glisser {
  0%   { opacity: 0; left: -50px; }
  100% { opacity: 1; left: 0; }
}
.titre {
  animation: glisser 1s ease 0.5s infinite;
}
transform

Applique un effet de transformation 2D ou 3D.

/* Rotation */
.carte:hover { transform: rotate(5deg); }
/* Agrandir */
img:hover    { transform: scale(1.1); }
/* Déplacer */
div          { transform: translate(50px, 20px); }
/* Incliner */
span         { transform: skew(10deg); }
filter

Définit des filtres visuels sur un élément.

/* Flou */
img { filter: blur(5px); }
/* Niveaux de gris */
img { filter: grayscale(100%); }
/* Inverser les couleurs */
img { filter: invert(1); }

Arithmétiques
let a = 10, b = 3;

a - b   // 7  (Soustraction)

Comparaison
5 == "5"  // true  (égal valeur)
5 === "5" // false (égal valeur ET type)
5 != 3    // true  (différent)
5 > 3     // true  (supérieur)
5 >= 5    // true  (supérieur ou égal)
5 < 3     // false (inférieur)
Logiques
// && = ET
(5 > 3) && (2 < 4)  // true

// || = OU
(5 > 10) || (2 < 4) // true

// ! = NON
!(5 > 3)             // false

Méthodes des chaînes de caractères

ch.length

Propriété qui retourne la longueur de la chaîne.

ch.indexOf()

Retourne la position de la 1ère occurrence d'une sous-chaîne (ou -1).

ch.substring()

Retourne une sous-chaîne de la position d à la position f (non incluse).

let ch = "Bonjour";

ch.substring(3)     // "jour"

Remplace la 1ère occurrence de ch1 par ch2.

let ch = "Bonjour Ali";

// Résultat: "Bonjour Ahmed"

Convertit tous les caractères en minuscules.

"BONJOUR".toLowerCase()
// "bonjour"
ch.toUpperCase()

Convertit tous les caractères en majuscules.

"bonjour".toUpperCase()
// "BONJOUR"
ch.trim()

Supprime les espaces au début et à la fin.

"  bonjour  ".trim()
// "bonjour"
String.fromCharCode()

Retourne une chaîne formée par la concaténation des codes ASCII.

String.fromCharCode(65)     // "A"

Fonctions de Date/Heure

date()

Formate une date et la retourne sous forme de chaîne.

date("d/m/Y")      // "29/04/2025"
date("H:i:s")      // "14:30:00"
date("Y-m-d H:i")  // "2025-04-29 14:30"
checkdate()

Vérifie la validité d'une date.

checkdate(2, 29, 2024)  // true (2024 bissextile)
checkdate(2, 29, 2023)  // false
strtotime()

Transforme une chaîne contenant une date en timestamp.

$ts = strtotime("2025-01-01");
echo date("d/m/Y", $ts); // "01/01/2025"

Fonctions sur les tableaux

array()

Crée un tableau à partir d'une liste de valeurs.

$t = array(1, 2, 3);
count()

Retourne le nombre d'éléments d'un tableau.

count([1, 2, 3]) // 3

Fonctions PHP pour MySQL

mysqli_connect()

Établit une connexion à une base de données MySQL.

$conn = mysqli_connect(
  "localhost",   // serveur
  "root",        // utilisateur
  "mdp",         // mot de passe
  "ma_bdd"       // base de données
);
if(!$conn) die("Erreur connexion!");
mysqli_query()

Envoie une requête SQL à la base de données.

$req = "SELECT * FROM eleves";
$result = mysqli_query($conn, $req);
mysqli_fetch_array()

Retourne une ligne de résultat sous forme de tableau associatif et numérique.

while($ligne = mysqli_fetch_array($result)) {
  echo $ligne["nom"];    // par nom de colonne
  echo $ligne[0];        // par index
}
mysqli_num_rows()

Retourne le nombre de lignes dans un résultat de requête.

$nb = mysqli_num_rows($result);
echo "Il y a $nb élèves.";
mysqli_close()

Ferme la connexion à la base de données.

mysqli_close($conn);

Variables superglobales et fonctions diverses

$_GET

Tableau des valeurs passées via les paramètres d'URL.

// URL: page.php?nom=Ali&age=18
$nom = $_GET["nom"];  // "Ali"
$age = $_GET["age"];  // "18"
$_POST

Tableau des valeurs passées via méthode POST (formulaire).

// Formulaire avec method="post"
$nom = $_POST["nom"];
$pwd = $_POST["password"];
isset()

Vérifie si une variable est définie.

if(isset($_POST["nom"])) {
  echo $_POST["nom"];
}
echo

Affiche une ou plusieurs expressions.

echo "Bonjour !";
echo "Résultat: " . $val;
echo "<p>$nom</p>";
require()

Inclut et exécute le fichier spécifié en argument.

require("connexion.php");
require("header.php");

Définition des données (DDL)

CREATE TABLE

Crée une nouvelle table dans la base de données.

CREATE TABLE eleves (
  id   INT PRIMARY KEY AUTO_INCREMENT,
  nom  VARCHAR(50) NOT NULL,
  age  INT CHECK(age >= 0),
  email VARCHAR(100) UNIQUE
);
ALTER TABLE

Modifie la structure d'une table existante.

-- Ajouter une colonne
ALTER TABLE eleves ADD telephone VARCHAR(15);

-- Supprimer une colonne
ALTER TABLE eleves DROP telephone;

-- Modifier le type
ALTER TABLE eleves MODIFY age DECIMAL;

-- Renommer
ALTER TABLE eleves RENAME age TO annee;
DROP TABLE

Supprime une table entièrement.

DROP TABLE eleves;

Contraintes d'intégrité

PRIMARY KEY

Définit la clé primaire (identifiant unique de la table).

CREATE TABLE eleves (
  id INT PRIMARY KEY,
  nom VARCHAR(50)
);
-- ou
CONSTRAINT pk PRIMARY KEY(id)
FOREIGN KEY / REFERENCES

Définit une clé étrangère référençant une autre table.

CREATE TABLE notes (
  id_eleve INT,
  note DECIMAL(4,2),
  FOREIGN KEY(id_eleve) 
    REFERENCES eleves(id)
    ON DELETE CASCADE
    ON UPDATE CASCADE
);
NOT NULL / DEFAULT / UNIQUE / CHECK

Contraintes de validation sur les valeurs des colonnes.

CREATE TABLE produits (
  id   INT PRIMARY KEY,
  nom  VARCHAR(50) NOT NULL,
  prix DECIMAL DEFAULT 0,
  code VARCHAR(10) UNIQUE,
  qte  INT CHECK(qte >= 0)
);

Manipulation des données (DML)

SELECT

Interroge la base de données. La requête la plus utilisée.

-- Tout sélectionner
SELECT * FROM eleves;

-- Colonnes spécifiques
SELECT nom, age FROM eleves;

-- Avec condition
SELECT * FROM eleves WHERE age > 17;

-- Tri
SELECT * FROM eleves ORDER BY nom ASC;

-- Groupe et filtre
SELECT classe, AVG(note) AS moyenne
FROM notes
GROUP BY classe
HAVING AVG(note) > 10;
INSERT INTO

Insère de nouvelles lignes dans une table.

INSERT INTO eleves(nom, age, email)
VALUES ("Ali Ben Salem", 18, "ali@mail.com");

-- Plusieurs lignes
INSERT INTO eleves(nom, age)
VALUES ("Sara", 17), ("Ahmed", 19);
UPDATE

Modifie les valeurs des colonnes dans une table.

UPDATE eleves
SET age = 19, email = "new@mail.com"
WHERE id = 5;

-- ⚠️ Sans WHERE : modifie TOUT!
UPDATE eleves SET age = 18;
DELETE FROM

Supprime des enregistrements d'une table.

DELETE FROM eleves WHERE id = 5;

-- ⚠️ Sans WHERE : supprime TOUT!
DELETE FROM eleves;

Fonctions d'agrégation

AVG()

Retourne la moyenne.

SELECT AVG(note) FROM notes;
COUNT()

Retourne le nombre d'enregistrements.

SELECT COUNT(*) FROM eleves;
MAX()

Retourne le maximum.

SELECT MAX(note) FROM notes;
MIN()

Retourne le minimum.

SELECT MIN(note) FROM notes;
SUM()

Retourne la somme.

SELECT SUM(prix) FROM produits;

Fonctions sur les dates

NOW()

Date et heure courantes.

SELECT NOW();
DAY() / MONTH() / YEAR()

Extraire jour, mois ou année.

SELECT YEAR(naissance) FROM e;
DATEDIFF()

Différence entre deux dates en jours.

SELECT DATEDIFF(d1, d2);
ADDDATE()

Ajoute des jours à une date.

SELECT ADDDATE(NOW(), 7);

Opérateurs de comparaison et logiques

BETWEEN

Filtre entre deux valeurs d'un intervalle.

SELECT * FROM eleves
WHERE age BETWEEN 15 AND 20;
LIKE

Recherche selon un motif (% = n'importe quoi, _ = 1 caractère).

-- Commence par "A"
WHERE nom LIKE "A%"

-- Contient "ali"
WHERE nom LIKE "%ali%"

-- Exactement 5 caractères
WHERE code LIKE "_____"
IN

Filtre si la valeur est dans une liste.

SELECT * FROM eleves
WHERE classe IN ("1A", "2B", "3C");
IS NULL / IS NOT NULL

Filtre les valeurs NULL.

SELECT * FROM eleves
WHERE email IS NULL;

SELECT * FROM eleves
WHERE email IS NOT NULL;
